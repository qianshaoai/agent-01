## 3. 实施要点

### 3.1 `lib/trial-agents.ts` 配置文件

```ts
export type TrialAgent = {
  id: string;
  platform: "coze" | "dify" | "yuanqi" | "qingyan" | "openai";
  name: string;
  description: string;
  avatar: string;
  apiEndpoint: string;
  botId: string;
  apiToken: string;
};

const RAW_AGENTS: TrialAgent[] = [
  {
    id: "agent_001",
    platform: "coze",
    name: "体验版智能体",
    description: "用于体验智能体问答能力",
    avatar: "",
    apiEndpoint: "https://api.coze.cn/v3/chat",
    botId: process.env.TRIAL_AGENT_001_BOT_ID ?? "",
    apiToken: process.env.TRIAL_AGENT_001_API_TOKEN ?? "",
  },
];

/** 已配置（env 都填了）的 agent，对外可见 */
export const trialAgents: TrialAgent[] = RAW_AGENTS.filter((a) => {
  const ok = a.botId && a.apiToken;
  if (!ok) {
    console.warn(`[trial-agents] disabled "${a.id}" — missing botId or apiToken in env`);
  }
  return ok;
});

/**
 * 按 id 查找 RAW（含未配置的 agent）。
 * 用于 chat 接口区分两种错误：
 *   - id 不在配置中 → 404（agent 根本不存在）
 *   - id 存在但 env 缺失 → 500（agent 已声明但配置不全）
 */
export function getTrialAgentRaw(id: string): TrialAgent | null {
  return RAW_AGENTS.find((a) => a.id === id) ?? null;
}

/** 按 id 查找已配置的 agent，列表脱敏后给前端 */
export function getTrialAgent(id: string): TrialAgent | null {
  return trialAgents.find((a) => a.id === id) ?? null;
}
```

**关键点**：

- 启动期 warn 列出哪些 agent 被禁用
- `trialAgents` / `getTrialAgent`：仅对外暴露已配置的 agent（用于 GET 列表脱敏 + 正常 chat）
- `getTrialAgentRaw`：内部使用，让 chat 接口能区分"agent 不存在 → 404"和"agent 存在但缺 env → 500"

### 3.2 `app/api/trial/agents/route.ts`

```ts
import { NextRequest, NextResponse } from "next/server";
import { getPayloadFromRequest, requireTrialUser } from "@/lib/auth";
import { trialAgents } from "@/lib/trial-agents";

export async function GET(req: NextRequest) {
  const payload = await getPayloadFromRequest(req);
  const guard = requireTrialUser(payload);
  if (guard) return guard;

  if (trialAgents.length === 0) {
    return NextResponse.json(
      { error: "trial agents not configured", code: "INTERNAL_ERROR" },
      { status: 500 }
    );
  }

  // 脱敏：不返回 botId / apiToken
  const data = trialAgents.map(({ id, name, description, avatar }) => ({
    id, name, description, avatar,
  }));

  return NextResponse.json({ data });
}
```

### 3.3 `app/api/trial/chat/route.ts`

```ts
import { NextRequest, NextResponse } from "next/server";
import { getPayloadFromRequest, requireTrialUser } from "@/lib/auth";
import { getTrialAgentRaw } from "@/lib/trial-agents";
import { streamChat } from "@/lib/adapters";
import { db } from "@/lib/db";

export async function POST(req: NextRequest) {
  const startedAt = Date.now();
  const payload = await getPayloadFromRequest(req);
  const guard = requireTrialUser(payload);
  if (guard) return guard;

  const body = await req.json().catch(() => ({}));
  const agentId: string = body.agent_id ?? "";
  const message: string = body.message ?? "";

  if (!agentId || !message) {
    return NextResponse.json({ error: "agent_id 和 message 必填" }, { status: 400 });
  }

  // 先查 RAW，区分 "id 不存在" vs "id 存在但缺配置"
  const raw = getTrialAgentRaw(agentId);
  if (!raw) {
    return NextResponse.json({ error: `trial agent not found: ${agentId}` }, { status: 404 });
  }
  if (!raw.botId || !raw.apiToken) {
    return NextResponse.json(
      { error: `trial agent ${agentId} missing botId/apiToken`, code: "INTERNAL_ERROR" },
      { status: 500 }
    );
  }
  const agent = raw; // 至此 botId / apiToken 必非空

  const cfg = {
    platform: agent.platform,
    apiEndpoint: agent.apiEndpoint,
    apiKey: agent.apiToken,
    agentCode: agent.botId,
    modelParams: { bot_id: agent.botId } as Record<string, unknown>,
  };

  const encoder = new TextEncoder();
  const stream = new ReadableStream({
    async start(controller) {
      let ok = true;
      let errMsg: string | undefined;
      try {
        for await (const chunk of streamChat([{ role: "user", content: message }], cfg)) {
          controller.enqueue(encoder.encode(`data: ${JSON.stringify({ delta: chunk })}\n\n`));
        }
        controller.enqueue(encoder.encode(`data: ${JSON.stringify({ done: true })}\n\n`));
      } catch (e: unknown) {
        ok = false;
        errMsg = e instanceof Error ? e.message : String(e);
        controller.enqueue(encoder.encode(`data: ${JSON.stringify({ error: errMsg })}\n\n`));
      } finally {
        controller.close();
        // 落 logs（非阻塞）
        const phone = payload && payload.type === "user" ? payload.phone : null;
        db.from("logs").insert({
          user_phone: phone,
          tenant_code: "PERSONAL",
          agent_code: agent.id,
          agent_name: agent.name,
          action: "trial_chat",
          status: ok ? "success" : "error",
          duration_ms: Date.now() - startedAt,
          error_msg: errMsg ?? null,
        }).then(() => {}, () => {});
      }
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache, no-transform",
      "Connection": "keep-alive",
    },
  });
}
```

**关键点**：

- 入口 `requireTrialUser`，非 trial 直接走 helper 兜底
- 第三方调用前再次校验 `botId / apiToken`，双保险
- SSE 协议：`data: {json}\n\n`，前端按 `data:` 行解析
- 不写 message 文本到 logs（合规 + 噪音少）

## 6. 风险与缓解

| 风险 | 缓解 |
|------|------|
| `streamChat` 抛错被 ReadableStream 吞掉 | try / catch 内部捕获并 enqueue 一条 `data: {error}` 给前端 |
| Coze 平台返回的字段需要适配 | `streamChat` 已封装；不直接对接 Coze API，避免多平台分支扩散 |
| logs 写入失败导致接口异常 | logs insert 用 `.then(noop, noop)` 异步丢弃错误，不影响响应 |
| SSE 在 Vercel 上被 buffer | response header 加 `Cache-Control: no-cache, no-transform`；本期先在本地验证，部署后另行回归 |
| 业务方提供的不是 Coze 智能体（如清言 / 元器 / Dify）| 当前 `cfg.modelParams` 只放了 `bot_id`（Coze 用）。其他平台需要补对应字段：清言（qingyan）通常需要 `assistant_id` 和单独的 `api_secret`；元器 需要 `assistant_id`；Dify 鉴权直接走 `apiKey`。本期只验证 Coze 路径；切换平台时需扩展 `TrialAgent` 类型加 `apiSecret`、`assistantId`，并在 chat route 构造 `cfg` 处按 platform 分支映射。**业务方明确平台后再动手**。 |
